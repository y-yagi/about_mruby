### H2O/mruby

```
# h2o.conf(H2Oの設定ファイル)
hosts:
  default:
    listen:
      port: 8080
    paths:
      /hello:
        mruby.handler-file: /etc/h2o/scripts/hello.rb
      /:
        file.dir: /var/www/html
        file.dirlisting: ON

access-log: /dev/stdout
error-log: /dev/stderr
```

```ruby
# /etc/h2o//scripts/hello.rb

Proc.new do |env|
  # HTTP status, Headers, Body
  [200, {'content-type' => 'text/plain'}, ["Hello world\n"]]
end
```
